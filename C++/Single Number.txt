class Solution {
public:
    int singleNumber(vector<int>& nums) {
       int i,j,c=0,n=0;
        for(i=0; i<nums.size();i++)
        {
            c=0;
            for(j=0;j<nums.size();j++)
            {
               if(nums[i]==nums[j])
               {
                  c++; 
               }
            }
            if(c==1)
            {
                return nums[i];
            }
                
        }
        return 1;
    }
};
